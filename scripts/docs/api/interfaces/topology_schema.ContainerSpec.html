    Interface ContainerSpec

    Normalized container specification used by services and sidecars

    Remarks

    Collection fields are fully normalized and defaults are applied before resolution

    interface ContainerSpec {
        buildContext?: string;
        command?: string[];
        configFiles?: Record<string, string>;
        dependsOn: string[];
        diagnostics: (
            | {
                authSecret?: string;
                headers?: Record<string, string>;
                name?: string;
                type?: "http";
                url?: string;
            }
            | { name?: string; serviceName?: string; type?: "dockerLogs" }
            | { name?: string; serviceName?: string; type?: "dockerInspect" }
        )[];
        dockerfile?: string;
        entrypoint?: string[];
        env: Record<string, string>;
        health?:
            | {
                headers?: Record<string, string>;
                path?: string;
                timeoutSeconds?: number;
                type?: "http";
                url?: string;
            }
            | { serviceName?: string; timeoutSeconds?: number; type?: "docker" }
            | { customTest?: string[]; timeoutSeconds?: number; type?: "command" };
        image?: string;
        ports: { container?: string; host?: string; protocol?: "tcp" | "udp" }[];
        resources?: { cpu?: string; localDisk?: string; memory?: string };
        secrets: string[];
        source?: {
            buildArgs?: Record<string, string>;
            dockerDir?: string;
            dockerfile?: string;
            dockerfileOverlay?: string;
            overlayDir?: string;
            ref?: string;
            repo?: string;
        };
        volumes: {
            kind?: "named"
            | "bind";
            mode?: string;
            source?: string;
            target?: string;
        }[];
    }
